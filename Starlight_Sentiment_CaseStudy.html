    # Try reading the CSV file with 'utf-8' encoding

    try:
        # If 'utf-8' fails, try 'utf-8-sig'
        df = pd.read_csv(file_path, encoding='utf-8-sig')

        # As a last resort, try 'ISO-8859-1' (also known as 'latin1')
        df = pd.read_csv(file_path, encoding='ISO-8859-1')

import nltk
from nltk.sentiment import SentimentIntensityAnalyzer
from nltk.tokenize import word_tokenize

#nltk.download('vader_lexicon')
#nltk.download('punkt')

positive_words = []
negative_words = []
neutral_words = []

def analyze_sentiment(review_text):
    """
    Analyzes the sentiment of a given drink review text and returns whether the sentiment
    is positive, negative, or neutral based on the compound score.
    """
    sia = SentimentIntensityAnalyzer()
    score = sia.polarity_scores(review_text)['compound']

    tokenized_text = word_tokenize(review_text)
    
    for word in tokenized_text:
        score = sia.polarity_scores(word)['compound']
        if score > 0.05:
            positive_words.append(word)
        elif score < -0.05:
            negative_words.append(word)
        else:
            neutral_words.append(word)

    score = sia.polarity_scores(review_text)['compound']
    
    if score > 0.05:
        return 'Positive'
    elif score < -0.05:
        return 'Negative'
    else:
        return 'Neutral'

import matplotlib.pyplot as plt

colors = {
    "Positive": "#4CAF50",  # Green
    "Neutral": "#FFC107",  # Amber
    "Negative": "#F44336",  # Red
}

# Filtering the dataset to include only the days from February 21 onwards for February and March
start_date = pd.to_datetime("2022-02-21", utc=True)
# Filter for dates starting from February 21
feb_march_from_21_daily_df = df[df['createdAt'] >= start_date]

# Group by date and sentiment, then count occurrences
grouped_from_21_daily = feb_march_from_21_daily_df.groupby([feb_march_from_21_daily_df['createdAt'].dt.date, 'sentiment']).size().unstack(fill_value=0)

# Calculate proportions
grouped_from_21_daily_prop = grouped_from_21_daily.div(grouped_from_21_daily.sum(axis=1), axis=0)

# Convert proportions to a "long" format for easier plotting
grouped_from_21_daily_prop_long = grouped_from_21_daily_prop.reset_index().melt(id_vars='createdAt', value_name='proportion')

# Plotting for daily data starting from February 21
fig, ax = plt.subplots(figsize=(12, 7))
for sentiment, color in colors.items():
    subset = grouped_from_21_daily_prop_long[grouped_from_21_daily_prop_long['sentiment'] == sentiment]
    ax.plot(pd.to_datetime(subset['createdAt']), subset['proportion'], marker='o', label=sentiment, color=color, linewidth=2)

ax.set_xlabel('Date')
ax.set_ylabel('Proportion')
ax.set_title('Daily Sentiment Proportions Starting February 21 (Colorful)')
plt.xticks(rotation=45)
plt.legend(title='Sentiment')
plt.grid(True, which='both', linestyle='--', linewidth=0.5)
plt.tight_layout()
plt.show()
